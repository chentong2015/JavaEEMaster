public static XmlDocument createXmlDocument(XmlSessionID xmlSessionID, String pkgName) throws Exception {
   XmlRequest xmlRequest = new XmlRequest(XmlRequest.strREQUEST_TYPE_SESSION, "Standard",
      AssemblyConstants.GET_ASSEMBLY_EXECUTION_CONTEXTS, xmlSessionID);
   GetAssemblyExecutionContextsRequest getAssemblyExecutionContextsRequest =
      new GetAssemblyExecutionContextsRequest(pkgName);
   XmlWriter xmlWriter = new XmlWriter();
   xmlRequest.toXml(xmlWriter, getAssemblyExecutionContextsRequest);
   return XmlDocument.create(xmlWriter);
}
